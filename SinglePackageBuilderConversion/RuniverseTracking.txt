Detecting “real new builds” externally

Workflow runs endpoint:
GET https://api.github.com/repos/r-universe/tempbioc/actions/workflows/build.yml/runs

Each run has:
  id (unique per run)
  head_sha (commit that triggered the run)
  event (push, workflow_dispatch, etc.)
  status (completed, queued, in_progress)
  conclusion (success, failure, cancelled)
  other??
head_sha should be different if a new build ran?


Version comparison
GET /repos/r-universe/tempbioc/contents/<package>/DESCRIPTION?ref=<head_sha>

parse the description? or from run name?? 



Architecture for external monitoring

Because you cannot run code inside R-universe:

    Monitoring script runs in your repo (or a CI you control):

        Polls the R-universe build workflow API every few minutes, or is triggered by a webhook if available.

        Checks head_sha vs last recorded SHA.

        Downloads DESCRIPTION at that SHA.

        Compares version to previously recorded version.

        Updates the stored version/commit info.

        Posts notifications if a new build/version is detected.

##  scripts/track_runiverse_build.py

import os
import csv
import requests
from datetime import datetime, timedelta
import subprocess

# ----------------------------
# Environment
# ----------------------------
GITHUB_TOKEN = os.environ["GITHUB_TOKEN"]
SUBMISSIONS_FILE = os.environ.get("SUBMISSIONS_FILE", "submissions/submitted_packages.csv")
RUNIVERSE_WORKFLOW = os.environ["RUNIVERSE_WORKFLOW"]
PACKAGE_NAME = os.environ.get("PACKAGE_NAME")
ISSUE_NUMBER = os.environ.get("ISSUE_NUMBER")

HEADERS = {
    "Authorization": f"Bearer {GITHUB_TOKEN}",
    "Accept": "application/vnd.github+json"
}

# ----------------------------
# 10-minute cutoff
# ----------------------------
cutoff_dt = datetime.utcnow() - timedelta(minutes=10)

# ----------------------------
# Load submissions
# ----------------------------
submissions = []
if os.path.exists(SUBMISSIONS_FILE):
    with open(SUBMISSIONS_FILE, newline="") as f:
        reader = csv.DictReader(f)
        for row in reader:
            row.setdefault("last_sha", "")
            row.setdefault("last_version", "")
            submissions.append(row)

# ----------------------------
# Version helpers
# ----------------------------
def parse_version(ver):
    try:
        x, y, z = [int(p) for p in ver.split(".")]
        return x, y, z
    except Exception:
        return None, None, None

def valid_z_bump(old, new):
    old_x, old_y, old_z = parse_version(old)
    new_x, new_y, new_z = parse_version(new)
    if old_x is None or new_x is None:
        return False
    return old_x == new_x and old_y == new_y and new_z > old_z

# ----------------------------
# Fetch latest workflow runs
# ----------------------------
def get_latest_runs(package):
    """Fetch workflow runs filtered for completed push events and 10-min cutoff."""
    url = RUNIVERSE_WORKFLOW.replace(
        "workflows/build.yml", 
        f"runs?branch=devel&event=push&status=completed&per_page=100"
    )
    resp = requests.get(url, headers=HEADERS)
    if resp.status_code != 200:
        print(f"[WARN] Failed to fetch workflow runs: {resp.text}")
        return []

    all_runs = resp.json().get("workflow_runs", [])
    filtered_runs = []
    seen_shas = set()
    for run in all_runs:
        run_time = datetime.strptime(run["created_at"], "%Y-%m-%dT%H:%M:%SZ")
        if run_time < cutoff_dt:
            continue
        sha = run.get("head_sha")
        if sha in seen_shas:
            continue
        seen_shas.add(sha)

        # Extract version from commit message: "Version: x.y.z"
        msg = run.get("head_commit", {}).get("message", "")
        version = msg.split("Version:")[-1].strip() if "Version:" in msg else None
        filtered_runs.append({
            "package": package,
            "sha": sha,
            "version": version,
            "run_url": run["html_url"]
        })
    return filtered_runs

# ----------------------------
# Update submissions
# ----------------------------
updated_rows = []
changes_made = False

for row in submissions:
    pkg = row["package_name"]
    last_sha = row.get("last_sha", "")
    last_version = row.get("last_version", "")

    if PACKAGE_NAME and pkg != PACKAGE_NAME:
        updated_rows.append(row)
        continue

    runs = get_latest_runs(pkg)
    if not runs:
        updated_rows.append(row)
        continue

    # Only consider the latest run
    run = runs[0]
    sha = run["sha"]
    version = run["version"]
    run_url = run["run_url"]

    if not sha or not version:
        updated_rows.append(row)
        continue

    if sha == last_sha and version == last_version:
        # No change, skip
        updated_rows.append(row)
        continue

    if valid_z_bump(last_version, version):
        print(f"[INFO] {pkg}: New build detected {last_version} -> {version}")
        row["last_sha"] = sha
        row["last_version"] = version
        changes_made = True

        # Post update to issue
        issue_num = row.get("issue_number")
        if issue_num:
            owner, repo = row["repo_full"].split("/")
            url = f"https://api.github.com/repos/{owner}/{repo}/issues/{issue_num}/comments"
            requests.post(url, headers=HEADERS, json={
                "body": f"✅ New build detected for {pkg}, version {version}.\n"
                        f"🔗 Detailed run: {run_url}\n"
                        f"📊 Check summary table: https://tempbioc.r-universe.dev/{pkg}$checktable"
            })
    else:
        # Invalid version bump
        print(f"[WARN] {pkg}: Version bump invalid ({last_version} -> {version})")
        issue_num = row.get("issue_number")
        if issue_num:
            owner, repo = row["repo_full"].split("/")
            url = f"https://api.github.com/repos/{owner}/{repo}/issues/{issue_num}/comments"
            requests.post(url, headers=HEADERS, json={
                "body": f"⚠️ Build detected for {pkg} with invalid version bump ({last_version} -> {version}). "
                        f"Only z should increase; please correct version.\n"
                        f"🔗 Detailed run: {run_url}\n"
                        f"📊 Check summary table: https://tempbioc.r-universe.dev/{pkg}$checktable"
            })
    updated_rows.append(row)

# ----------------------------
# Commit updated CSV if needed
# ----------------------------
if changes_made:
    branch = "submissions"
    subprocess.run(["git", "fetch", "origin", branch])
    subprocess.run(["git", "checkout", "-B", branch, f"origin/{branch}"])

    fieldnames = ["package_name","repo_full","submitter","issue_number","last_sha","last_version"]
    with open(SUBMISSIONS_FILE, "w", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        writer.writerows(updated_rows)

    subprocess.run(["git", "config", "user.name", "Bioconductor Bot"])
    subprocess.run(["git", "config", "user.email", "bot@bioconductor.org"])
    subprocess.run(["git", "add", SUBMISSIONS_FILE])
    subprocess.run(["git", "commit", "-m", "Update build SHAs and versions"])
    subprocess.run(["git", "push", "origin", branch])
else:
    print("[INFO] No updates detected; nothing to commit.")
